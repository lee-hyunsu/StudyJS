------------------------------NaN 값 비교--------------------------------


<!DOCTYPE html>
<html lang="en">
<head>
    <title>hyunsu</title>
    <script>
        if(NaN == NaN){
            alert('NaN == NaN');
        }else{
            alert('NaN != NaN');
        }
    </script>
</head>
<body>
    hyunsu
</body>
</html>



------------------------------isNaN() 함수--------------------------------------



<!DOCTYPE html>
<html lang="en">
<head>
    <title>hyunsu</title>
    <script>
        if(isNaN(NaN)){
            alert('NaN == NaN')
        }else{
            alert('NaN != NaN');
        }
    </script>
</head>
<body>
    hyunsu
</body>
</html>
